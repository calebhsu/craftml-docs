```craftml {useWorker:false}
<craft>
    <script type="text/craftml">
        function main(){
            var data = [20, 14, 50, 23, 30, 32, 12, 35, 16, 9, 10]

            var width = 40,
                barHeight = 5

            var x = d3.scale.linear()
                .range([0, width])
                .domain([0, d3.max(data, function(d) {
                    return d
                })])

            d3.select("body")
                    .select("craft")
                    .remove()

            d3.select("body")
                .append('craft')
                .selectAll("cube")
                .data(data)
                .enter()
                .append("cube")
                .attr("x", function(d,i){
                    return i * 6
                })
                .attr("xsize", barHeight)
                .attr("ysize", function(d) {

                    return x(d)
                })

            var xml = d3.select("craft").html()
            return xml
        }
    </script>
</craft>

```
